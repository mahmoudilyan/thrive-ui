Usage Example

<!-- Include CSS -->
<link rel="stylesheet" href="vanilla-select-multi-category.css">

<!-- Container element -->
<div id="my-select"></div>

<!-- Include JS -->
<script src="vanilla-select-multi-category.js"></script>

<script>
  const container = document.getElementById('my-select');
  const select = new SelectMultiCategory(container, {
    categories: [
      {
        name: 'Fruits',
        items: [
          { value: 'apple', label: 'Apple', description: 'A red fruit' },
          { value: 'banana', label: 'Banana' }
        ]
      }
    ],
    placeholder: 'Select items...',
    searchPlaceholder: 'Search...',
    onChange: (values) => {
      console.log('Selected:', values);
    }
  });
</script>
